import itertools

standards_funcs = set([ x[0] for x in standards_funcs[1:]])
rfc_funcs = set([ x[0] for x in rfc_funcs])
covered_by_rfc = sorted(standards_funcs.intersection(rfc_funcs))
not_covered_by_rfc = sorted(standards_funcs.difference(rfc_funcs))
in_rfc_not_in_std = sorted(rfc_funcs.difference(standards_funcs))
def prt(x):
  for func in x:
    print("\t", func)

print("Percent Coverage: {:.1f}%".format(len(covered_by_rfc) * 100 / len(standards_funcs)))

print("Functions covered by RFC notes:")
prt(covered_by_rfc)
print("Functions not covered by RFC notes:")
prt(not_covered_by_rfc)
print("Functions in RFC notes but not in Standard:")
prt(in_rfc_not_in_std)
